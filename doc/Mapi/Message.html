def inspect
	str = %w[message_class from to subject].map do |key|
		" #{key}=#{props.send(key).inspect}"
	end.compact.join
	str << " recipients=#{recipients.inspect}"
	str << " attachments=#{attachments.inspect}"
	"#<#{self.class.to_s[/\w+$/]}#{str}>"
end
